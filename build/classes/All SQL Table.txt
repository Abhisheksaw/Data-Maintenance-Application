CREATE TABLE BRANCHS(
  BRANCH_ID VARCHAR2(10) CONSTRAINT BRANCH_PK PRIMARY KEY,
  BRANCH_NAME VARCHAR2(100) CONSTRAINT BRA_UNQ_NAME UNIQUE,
  ADMIN_NAME VARCHAR2(50),
  PASSWORD VARCHAR2(20),
  STATUS VARCHAR2(50),
  EMAIL_ID VARCHAR2(100) CONSTRAINT BRA_UNQ_EM_ID UNIQUE,
  SECURITY_KEY VARCHAR2(100)
  );

CREATE TABLE USERS( 
 USER_ID VARCHAR2(10) CONSTRAINT USER_PK PRIMARY KEY,  
 BRANCH_ID VARCHAR2(30) CONSTRAINT USR_FK_CMP_ID REFERENCES
 BRANCHS(BRANCH_ID),
 EMAIL_ID VARCHAR2(30) CONSTRAINT USR_NN_EMID NOT NULL, 
 PASSWORD VARCHAR2(30) CONSTRAINT USR_NN_PWD NOT NULL, 
 USER_NAME VARCHAR2(50) 
);

CREATE TABLE STUDENTS(
 STUDENT_ID VARCHAR2(20) PRIMARY KEY NOT NULL,
 NAME VARCHAR2(20) NOT NULL,
 DOB DATE NOT NULL,
 GENDER VARCHAR2(10) NOT NULL,
 EMAIL VARCHAR2(50) NOT NULL,
 PHONE_NO VARCHAR2(15) NOT NULL,
 FATHER_NAME VARCHAR2(20) NOT NULL,
 MOTHER_NAME VARCHAR2(20) NOT NULL,
 LOCAL_ADDRESS VARCHAR2(200) NOT NULL,
 PERMANENT_ADDRESS VARCHAR2(200) NOT NULL,
 IMAGE_PATH VARCHAR2(300) NOT NULL 
);

4.

CREATE TABLE COURSES(
  STUDENT_ID VARCHAR2(20) PRIMARY KEY NOT NULL,
  ID VARCHAR2(20) DEFAULT NULL,
  SEMESTER INT DEFAULT NULL,
  COURSE1 VARCHAR2(200) DEFAULT NULL,
  COURSE2 VARCHAR2(200) DEFAULT NULL,
  COURSE3 VARCHAR2(200) DEFAULT NULL,
  COURSE4 VARCHAR2(200) DEFAULT NULL,
  COURSE5 VARCHAR2(200) DEFAULT NULL,
  CONSTRAINT fk_ID FOREIGN KEY (ID) REFERENCES STUDENTS (STUDENT_ID) ON DELETE CASCADE
  );

5.

CREATE TABLE SCORES(
  STUDENT_ID VARCHAR2(20) PRIMARY KEY NOT NULL,
  ID VARCHAR2(20) NOT NULL,
  SEMESTER INT NOT NULL,
  COURSE1 VARCHAR(200) NOT NULL,
  SCORE1 VARCHAR(5) NOT NULL,
  COURSE2 VARCHAR(200) NOT NULL,
  SCORE2 VARCHAR(5) NOT NULL,
  COURSE3 VARCHAR(200) NOT NULL,
  SCORE3 VARCHAR(5) NOT NULL,
  COURSE4 VARCHAR(200) NOT NULL,
  SCORE4 VARCHAR(5) NOT NULL,
  COURSE5 VARCHAR(200) NOT NULL,
  SCORE5 VARCHAR(5) NOT NULL,
  AVERAGE VARCHAR(5) NOT NULL,
  CONSTRAINT fk_STU_ID FOREIGN KEY (ID) REFERENCES STUDENTS (STUDENT_ID) ON DELETE CASCADE
  );
